`timescale 1ns / 1ps

module gray_tb(

    );
    
    reg x1,x2,x3,x4;
    wire y1,y2,y3,y4;
    
    gray dut (x1,x2,x3,x4,y1,y2,y3,y4);
    
    initial 
      begin: stimulus
        reg [3:0] i;
        for (i = 0 ; i < 16 ; i = i + 1)
        begin
          {x1,x2,x3,x4} = i [3:0];
          #1 $display ("x1x2x3x4 = %b , z1z2z3z4 = %b", {x1,x2,x3,x4} , {y1,y2,y3,y4});
        end 
      #1 $stop;
      end
       
endmodule
